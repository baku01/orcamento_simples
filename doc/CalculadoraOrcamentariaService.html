def valor_venda_eqp(
  valor_hora:,
  despesa_indireta:,
  margem_lucro_fixa:,
  imposto:,
  quantidade_horas:
)
  (valor_hora * (despesa_indireta / (1 - (margem_lucro_fixa + imposto)))) * quantidade_horas
end

def valor_venda_mo(valor_hora:, valor_base:, taxa_variacao:, quantidade_horas:)
